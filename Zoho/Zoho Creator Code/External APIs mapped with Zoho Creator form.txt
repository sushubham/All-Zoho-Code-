void apiWithZohoCreator(int recid)
{
    empinfo = Employee[ID == recid];//find out duplicate value or record

    headermap = Map();
    headermap.put("Token", "285ee051e8be4d92aaa63f4445a41401");
    headermap.put("Content-Type", "application/json");
    paramp = Map();
	paramp.put("badgeNumber", empinfo.BadgeNumber);
    response = invokeurl
    [
        url : apiUrl
        type : POST
        parameters : paramp.toString()
        headers : headermap
    ];
//     info "API Response: " + response; //GIVES NULL VALUE
    resultList = response.get("message"); //get message
// 	info resultList.toJSONList();
	for each params in resultList.toJSONList() //convert into JSON Object
	{
		idValue = params.get("Id");
// 		info idValue;
		existing = Attendance_Entries[Id1 == idValue];
		if (existing.ID.isEmpty())
		{
			attendata = insert into Attendance_Entries
			[
				Added_User = zoho.loginuser
				Id1 = idValue
				VerifyType = params.get("VerifyType").toString()

				StatusId = params.get("StatusId")
				DeviceSerialNumber = params.get("DeviceSerialNumber").toString()
				DeviceAliasName = params.get("DeviceAliasName").toString()
			];
		}
	}

}
